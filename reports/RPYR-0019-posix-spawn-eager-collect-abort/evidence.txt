# RustPython 0.5.0 (a9c2c529b) vs CPython 3.14.4.

# --- cheap type divergence (confirmed by synthesizer) ---
$ rustpython -c "import os; os.waitpid(os.posix_spawn('/bin/true',(str(x) for x in range(3)),os.environ),0)"
                                    # accepts + materializes the generator, spawns -> ran (exit 0)
$ python3 -c "import os; os.posix_spawn('/bin/true',(str(x) for x in range(3)),os.environ)"

# --- OOM abort on an infinite iterable (eager-collect-parity auditor, live) ---

$ python3  -c "import os, itertools; os.posix_spawn('/bin/true', map(str, itertools.count()), {})"
